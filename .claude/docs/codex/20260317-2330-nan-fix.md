```python
def angle_loss(pred_params, gt_params, confidence, valid_mask):
    p = pred_params[..., 0]; g = gt_params[..., 0]
    finite = valid_mask & torch.isfinite(p) & torch.isfinite(g) & torch.isfinite(confidence)
    if not finite.any():
        return pred_params.new_zeros(())
    idx = finite.nonzero(as_tuple=True)
    p, g, w = p[idx], g[idx], confidence[idx]
    dot = torch.cos(p) * torch.cos(g) + torch.sin(p) * torch.sin(g)
    loss = 1.0 - dot.abs()
    return ((loss * w).sum() / (w.sum() + 1e-8)) * 0.01
```
